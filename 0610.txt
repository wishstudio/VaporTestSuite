class Parent
{
	public virtual function GetNumber() -> int32
	{
		return 10;
	}
}

class Child: Parent
{
	public override function GetNumber() -> int32
	{
		return 20;
	}
}

class Test
{
	public static function Main()
	{
		Parent obj1 = new Parent(), obj2 = new Child();
		Core.Console::WriteLine(Core.Int32::ToString(obj1::GetNumber()));
		Core.Console::WriteLine(Core.Int32::ToString(obj2::GetNumber()));
	}
}
