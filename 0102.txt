	{
		Core.Console::WriteLine("Executed condition 1.");

	{
		Core.Console::WriteLine("Executed condition 2.");
		return 0;
	}

	public static function Condition3() -> int32
	{
		Core.Console::WriteLine("Executed condition 3.");
		return 1;

	public static function Main()
	{
		/* Priority of && is higher than || */

			Core.Console::WriteLine("Executed body 2.");
		else
			Core.Console::WriteLine("Not executed body 2.");

			Core.Console::WriteLine("Executed body 3.");
		else
			Core.Console::WriteLine("Not executed body 3.");
