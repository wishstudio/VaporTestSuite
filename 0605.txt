/*
Vapor Test Suite
0605: Overloaded instance constructors

- output
Hello world.
String
42
*/

class Test
{
	public Core.String text;

	public .ctor()
	{
		this::text = "Hello world.";
	}

	public .ctor(Core.String text)
	{
		this::text = text;
	}

	public .ctor(int32 num)
	{
		this::text = Core.Int32::ToString(num);
	}

	public static OutputTest(Test obj)
	{
		Core.Console::WriteLine(obj::text);
	}

	public static Main()
	{
		OutputTest(new Test());
		OutputTest(new Test("String"));
		OutputTest(new Test(42));
	}
}
