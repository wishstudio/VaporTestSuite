class Test
{
	public static Fibonacci(int32 n) -> int32
	{
		if (n == 0)
			return 0;
		else if (n == 1)
			return 1;
		else
			return Fibonacci(n - 1) + Fibonacci(n - 2);
	}

	public static Main()
	{
		Core.Console::WriteLine(Core.Int32::ToString(Fibonacci(10)));
	}
}
