class Test
{
	public static function Reverse(int32 a, int32 b) -> int32, int32
	{
		return b, a;
	}

	public static function Main()
	{
		int32 a = 42, b = 24;
		Core.Console::WriteLine(Core.Int32::ToString(a) + " " + Core.Int32::ToString(b));
		a, b = Reverse(a, b);
		Core.Console::WriteLine(Core.Int32::ToString(a) + " " + Core.Int32::ToString(b));
	}
}
