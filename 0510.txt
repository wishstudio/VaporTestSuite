/*
Vapor Test Suite
0510: Overloaded functions

- output
Func1.
Func2.True
Func2.False
Func3.42
Func4.Life
Func4...
*/

class Test
{
	public static function Func()
	{
		Core.Console::WriteLine("Func1.");
	}

	public static function Func(bool value)
	{
		if (value)
			Core.Console::WriteLine("Func2.True");
		else
			Core.Console::WriteLine("Func2.False");
	}

	public static function Func(int32 value)
	{
		Core.Console::WriteLine("Func3." + Core.Int32::ToString(value));
	}

	public static function Func(Core.String value)
	{
		if (value == null)
			Core.Console::WriteLine("Func4...");
		else
			Core.Console::WriteLine("Func4." + value);
	}

	public static function Main()
	{
		Func();
		Func(true);
		Func(2 > 3);
		Func(42);
		Func("Life");
		Func(null);
	}
}
