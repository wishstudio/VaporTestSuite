class Test
{
	public static function Func()
	{
		Core.Console::WriteLine("It is Func!");
	}

	public static function Func1(int32 num)
	{
		Core.Console::WriteLine("Parameter number is " + Core.Int32::ToString(num));
	}

	public static function Main()
	{
		Core.Console::WriteLine("Calling function...");
		Func();
		Func1(42);
		Core.Console::WriteLine("Called function.");
	}
}
